--- FILE: ./firestore.rules ---
rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {

    // ==========================================
    // 🛡️ HELPERS DE SEGURIDAD
    // ==========================================
    function isAuthenticated() { return request.auth != null; }
    function isOwner(userId) { return isAuthenticated() && request.auth.uid == userId; }
    function getUserData() { return get(/databases/$(database)/documents/users/$(request.auth.uid)).data; }
    
    function isStaff() {
      let role = getUserData().rol;
      return isAuthenticated() && (role == 'admin' || role == 'administrador' || role == 'profesional');
    }
    function isAdmin() {
      let role = getUserData().rol;
      return isAuthenticated() && (role == 'admin' || role == 'administrador');
    }

    // ==========================================
    // 📋 REGLAS POR COLECCIÓN
    // ==========================================

    // ✅ REGLA CRÍTICA PARA ACTIVACIÓN (Nuevos Usuarios)
    match /patient_data/{historyId} {
      // Permite verificar si el Nº de Historia existe antes de loguearse
      // Se permite 'get' (documento individual) pero NO 'list' (ver todos)
      allow get: if true; 
      allow list: if isStaff(); // Solo el staff puede ver la lista completa
      allow write: if isStaff();
    }

    match /users/{userId} {
      allow read: if isOwner(userId) || isStaff();
      allow create: if isOwner(userId); 
      allow update: if (isOwner(userId) 
                        && request.resource.data.diff(resource.data).affectedKeys()
                           .hasOnly(['nombre', 'apellidos', 'telefono', 'fechaNacimiento', 'codigoPostal', 'fotoUrl', 'fcmToken', 'termsAccepted', 'privacyAccepted', 'sesionesRestantes', 'consentDate', 'consentVersion', 'rol'])) 
                    || isStaff();
      allow delete: if isAdmin();
    }

    match /bookings/{bookingId} {
      allow read: if isOwner(resource.data.userId) || isStaff();
      allow create: if isAuthenticated() && request.resource.data.userId == request.auth.uid;
      allow delete: if isAuthenticated() && (resource.data.userId == request.auth.uid || isStaff());
      allow update: if isStaff();
    }

    match /appointments/{appointmentId} {
      allow read: if isOwner(resource.data.userId) || isStaff();
      allow write: if isStaff();
    }
    
    match /groupClasses/{classId} {
      allow read: if isAuthenticated();
      allow update: if isAuthenticated() && (
        isStaff() || 
        (request.resource.data.diff(resource.data).affectedKeys().hasOnly(['aforoActual']) && 
         (request.resource.data.aforoActual == resource.data.aforoActual + 1 || 
          request.resource.data.aforoActual == resource.data.aforoActual - 1))
      );
      allow create, delete: if isStaff();
    }

    match /passes/{passId} {
      allow read: if isAuthenticated() && (resource.data.userId == request.auth.uid || isStaff());
      allow update: if isAuthenticated() && (
        isStaff() || 
        (resource.data.userId == request.auth.uid && 
         request.resource.data.diff(resource.data).affectedKeys().hasOnly(['tokensRestantes']))
      );
      allow create, delete: if isStaff();
    }

    match /exercises/{exerciseId} { allow read: if isAuthenticated(); allow write: if isStaff(); }
    match /exercise_assignments/{assignId} {
      allow read: if isAuthenticated() && (resource.data.userId == request.auth.uid || isStaff());
      allow create, delete: if isStaff();
      allow update: if isStaff() || (isOwner(resource.data.userId) && request.resource.data.diff(resource.data).affectedKeys().hasOnly(['completado']));
    }

    match /documents/{docId} {
      allow read: if isAuthenticated() && (resource.data.userId == request.auth.uid || isStaff());
      allow create, update: if isStaff(); 
      allow delete: if isAdmin();
    }

    match /consent_templates/{templateId} { allow read: if isAuthenticated(); allow write: if isAdmin(); }
    match /legacy_import/{docId} { allow read: if isAuthenticated() && (resource.data.email == request.auth.token.email || isStaff()); allow write: if isStaff(); }

    match /timeClockRecords/{recordId} { allow read, write: if isStaff(); }
    match /internal_chats/{chatId} { 
      allow read, write: if isStaff(); 
      match /messages/{msgId} { allow read, write: if isStaff(); } 
    }
    match /internal_tasks/{taskId} { allow read, write: if isStaff(); }
    match /broadcasts/{broadcastId} { allow read: if isAuthenticated(); allow write: if isAdmin(); }
    match /journal/{entryId} { allow read: if isOwner(resource.data.userId) || isStaff(); allow create: if isOwner(request.resource.data.userId); allow update: if isStaff(); }
    match /metrics/{metricId} { allow read: if isOwner(resource.data.userId) || isStaff(); allow create: if isOwner(request.resource.data.userId); }
    match /audit_logs/{logId} { allow create: if isAuthenticated(); allow read: if isAdmin(); allow update, delete: if false; }
    match /system_config/{docId} { allow read: if isAuthenticated(); allow write: if isAdmin(); }
    match /config/{docId} { allow read: if true; allow write: if isAdmin(); }
  }
}

import * as functions from "firebase-functions";
import * as admin from "firebase-admin";

admin.initializeApp();
const db = admin.firestore();

export const checkAccountStatus = functions.https.onCall(async (data, context) => {
  const email = (data.email as string).trim().toLowerCase();
  const historyIdRaw = data.historyId.toString().trim();
  
  // 1. Normalización para comparación
  const idString = historyIdRaw.padStart(6, "0"); // "000001"
  const idNumber = parseInt(historyIdRaw, 10);   // 1
  
  try {
    // 2. Primero verificamos si ya existe en 'users'
    const userSnapshot = await db.collection("users")
      .where("email", "==", email)
      .limit(1)
      .get();
      
    if (!userSnapshot.empty) {
      return { status: "ALREADY_REGISTERED" };
    }
    
    // 3. Si no existe, buscamos en 'legacy_import' por EMAIL
    // Buscamos todos los registros con ese email (suele ser solo 1)
    const legacySnapshot = await db.collection("legacy_import")
      .where("email", "==", email)
      .get();
      
    if (legacySnapshot.empty) {
      return { status: "NOT_FOUND" };
    }
    
    // 4. Verificación HÍBRIDA de ID de historia
    // Buscamos en los documentos encontrados si alguno coincide con el ID (como string o como número)
    const match = legacySnapshot.docs.find(doc => {
      const d = doc.data();
      // Buscamos en los nombres de campo más probables
      const dbId = d.historyId || d.idH || d.numero;
      
      // Comparamos sin importar el tipo (== hace cast automático en JS/TS para esto)
      return dbId == idString || dbId == idNumber || dbId == historyIdRaw;
    });
    
    if (match) {
      return { status: "ACTIVATION_PENDING" };
    } else {
      return { status: "NOT_FOUND" };
    }
    
  } catch (error) {
    console.error("Error en checkAccountStatus:", error);
    throw new functions.https.HttpsError("internal", "Error al verificar la cuenta");
  }
});
